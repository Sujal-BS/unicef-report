import polars as pl
from plotnine import *

# 2. Load the data directly from your uploaded files
indicator_1 = pl.read_csv("unicef_indicator_1.csv")
metadata = pl.read_csv("unicef_metadata.csv")

# 3. Rename columns so they match for a clean join
ind_clean = indicator_1.rename({"time_period": "year"})
meta_clean = metadata.rename({"GDP per capita (constant 2015 US$)": "GDP_per_capita"})

# 4. The required Inner Join
df_merged = ind_clean.join(meta_clean, on=["country", "year"], how="inner")

# 5. Drop missing values to ensure charts render perfectly
df_clean = df_merged.drop_nulls(subset=["obs_value", "GDP_per_capita"])

# 6. Advanced Polars: Create an Income Group column for our scatterplot
df_clean = df_clean.with_columns(
    pl.when(pl.col("GDP_per_capita") > 20000).then(pl.lit("High Income"))
    .when(pl.col("GDP_per_capita") > 5000).then(pl.lit("Middle Income"))
    .otherwise(pl.lit("Low Income"))
    .alias("Income_Group")
)

print(f"Data cleaned successfully! Rows remaining: {df_clean.height}")

To perform a powerful review of the connection of national prosperity against world health results, the key child mortality measure was combined with the UNICEF metadata data-set. An inner join was done using both country and year variable as keys. This particular estimate was selected to ensure that all observations of mortality are matched with the exact, corresponding GDP per capita of the exact year or else, incorrect historical data will be involved.

After the merge, the rows that had both null values in either the mortality rate (obs_value) or the GDP less capital (capita) (in the columns) were strictly filtered. The nulls must be dropped in order to preserve data integrity so that only complete and verifiable data is used to calculate trends using the linear regression model. Finally, to enrich the Exploratory Data Analysis (EDA), an advanced Polars transformation was used to engineer a custom Income_Group feature. This classifies countries into High (>20,000), Middle (>5,000) and Low Income brackets which can serve as very important demographic background to the visual story.

import geopandas as gpd
import polars as pl
from plotnine import *

# 1. Force the cell to grab the data itself
ind_map = pl.read_csv("unicef_indicator_1.csv").rename({"time_period": "year"})
meta_map = pl.read_csv("unicef_metadata.csv").rename({"GDP per capita (constant 2015 US$)": "GDP_per_capita"})
df_map_clean = ind_map.join(meta_map, on=["country", "year"], how="inner").drop_nulls(subset=["obs_value", "GDP_per_capita"])

# Get the most recent year
max_year = df_map_clean["year"].max()

# 2. Get the world map from a direct, stable URL (bypasses the deprecated error)
world_url = "https://naturalearth.s3.amazonaws.com/110m_cultural/ne_110m_admin_0_countries.zip"
world = gpd.read_file(world_url)

# 3. Merge the data with the map (using ISO_A3 from the new dataset)
map_data = world.merge(df_map_clean.filter(pl.col("year") == max_year).to_pandas(),
                       left_on='ISO_A3',
                       right_on='alpha_3_code',
                       how='left')

# 4. Plot the map
(ggplot(map_data)
 + geom_map(aes(fill='obs_value'), color="white", size=0.2)
 + scale_fill_cmap(name="viridis", na_value="lightgrey")
 + theme_void()
 + labs(title="Global Child Mortality Rates", fill="Rate")
 + theme(figure_size=(10, 6))
)
